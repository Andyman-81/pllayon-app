STEP 1 — CREATE A TEST HOOK
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Create a shared hook: useLinkedAthlete()

This hook is used by every coach and parent component
that needs to access athlete data.

  export function useLinkedAthlete() {
    const [athleteId, setAthleteId] = useState(null)
    const [athleteData, setAthleteData] = useState(null)
    const [loading, setLoading] = useState(true)
    const [error, setError] = useState(null)

    useEffect(() => {
      async function fetchLinkedAthlete() {
        try {
          const { data: { user } } = await supabase.auth.getUser()
          if (!user) { setError('Not logged in'); return }

          const role = localStorage.getItem('po_role')

          if (role === 'coach') {
            // Get linked athlete via coach_athlete_links
            const { data: link, error: linkError } = await supabase
              .from('coach_athlete_links')
              .select('athlete_id')
              .eq('coach_id', user.id)
              .single()

            if (linkError || !link) {
              setError('no_athlete_linked')
              setLoading(false)
              return
            }

            const { data: athlete } = await supabase
              .from('athletes')
              .select('*')
              .eq('id', link.athlete_id)
              .single()

            setAthleteId(link.athlete_id)
            setAthleteData(athlete)

          } else if (role === 'parent') {
            // Get linked athlete via parent_athlete_links
            const { data: link, error: linkError } = await supabase
              .from('parent_athlete_links')
              .select('athlete_id')
              .eq('parent_id', user.id)
              .single()

            if (linkError || !link) {
              setError('no_athlete_linked')
              setLoading(false)
              return
            }

            const { data: athlete } = await supabase
              .from('athletes')
              .select('id, name, sport')
              .eq('id', link.athlete_id)
              .single()

            setAthleteId(link.athlete_id)
            setAthleteData(athlete)
          }
        } catch (e) {
          setError(e.message)
        } finally {
          setLoading(false)
        }
      }

      fetchLinkedAthlete()
    }, [])

    return { athleteId, athleteData, loading, error }
  }

Replace all existing coach and parent data fetching with this hook.
No coach or parent component should be querying athlete data
without going through useLinkedAthlete() first.

  function ProtectedRoute({ allowedRoles, children }) {
    const role = localStorage.getItem('po_role')
    const { data: { session } } = useSession() // or however auth is checked

    if (!session) return <Navigate to="/login" />
    if (allowedRoles && !allowedRoles.includes(role)) {
      return <Navigate to={`/dashboard/${role || 'athlete'}`} />
    }
    return children
  }

  // Usage in router:
  <Route path="/dashboard/coach" element={
    <ProtectedRoute allowedRoles={['coach']}>
      <CoachDashboard />
    </ProtectedRoute>
  } />
